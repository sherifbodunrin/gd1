C++ Code Example

This is the C++ code that reads resource allocation data from a file (input1.txt):


#include <iostream>
#include <string>
#include <fstream>
using namespace std;

bool issafe()
{
    return true;
}

int n;
int m;
int* available;
int** allocation;
int** max;

int main()
{
    string filename;
    ifstream datafile;

    // File input is directly referred as "input1.txt"
    datafile.open("input1.txt");

    if (!datafile.good())
    {
        cout << "File not opened correctly";
        return -1;
    }

    datafile >> n;
    datafile >> m;

    available = new int[m];
    for (int count = 0; count < m; count++)
        datafile >> available[count];

    cout << "Number of processes: " << n << endl;
    cout << "Number of types of resources: " << m << endl;

    for (int count = 0; count < m; count++)
        cout << available[count] << ", ";

    datafile.close();
    cout << issafe() << endl;

    delete[] available;

    return 0;
}

Note: The file input1.txt needs to be placed in the same directory as the C++ executable for it to be read correctly.
